🔐 نظام المصادقة 🔐 Authentication System

Bearer Token Authentication Bearer Token Authentication

جميع الطلبات المحمية تتطلب إضافة Bearer Token في header: All protected requests require Bearer Token in header: 

Authorization: Bearer {your_access_token}
📝 مثال كامل للمصادقة 📝 Complete Authentication Example 

// Flutter Example
class ApiService {
  static const String baseUrl = 'https://ahjili.com/api';
  static String? authToken;
  
  static Map<String, String> get headers => {
    'Content-Type': 'application/json',
    'Accept': 'application/json',
    if (authToken != null) 'Authorization': 'Bearer $authToken',
  };
}

Future<LoginResponse> login(String username, String password) async {
  final response = await http.post(
    Uri.parse('$baseUrl/auth/login'),
    headers: headers,
    body: jsonEncode({
      'username': username,
      'password': password,
      'remember': true,
    }),
  );
  
  if (response.statusCode == 200) {
    final data = jsonDecode(response.body);
    authToken = data['data']['token'];
    return LoginResponse.fromJson(data);
  } else {
    throw ApiException.fromResponse(response);
  }
}

📊 نماذج البيانات 📊 Data Models

👤 نموذج المستخدم (User) 👤 User Model 
interface User {
  id: number;
  username: string;
  display_name: string;
  email?: string; // Only in profile context
  avatar_url?: string;
  bio?: string;
  location?: string;
  role: 'user' | 'moderator' | 'admin' | 'super-admin';
  account_type: 'regular' | 'verified' | 'counselor';
  is_active: boolean;
  created_at: string; // ISO 8601
  last_seen_at?: string; // ISO 8601
}
📝 نموذج المنشور (Post) 📝 Post Model 
interface Post {
  id: number;
  content: string;
  image_url?: string;
  category: 'complaint' | 'experience' | 'recommendation' | 'question' | 'review' | 'general';
  type: 'anonymous' | 'community';
  location?: string;
  hashtags: string[];
  status: 'pending' | 'approved' | 'rejected';
  likes_count: number;
  comments_count: number;
  shares_count: number;
  views_count: number;
  created_at: string; // ISO 8601
  updated_at: string; // ISO 8601
  user?: User; // null for anonymous posts
  is_liked: boolean; // Current user's like status
  is_bookmarked: boolean; // Current user's bookmark status
}
💬 نموذج التعليق (Comment) 💬 Comment Model 
interface Comment {
  id: number;
  post_id: number;
  parent_id?: number; // For replies
  content: string;
  likes_count: number;
  replies_count: number;
  status: 'pending' | 'approved' | 'rejected';
  is_anonymous: boolean;
  anonymous_name?: string; // For anonymous comments
  created_at: string; // ISO 8601
  user?: User; // null for anonymous comments
  replies?: Comment[]; // Nested replies
  is_liked: boolean; // Current user's like status
}
